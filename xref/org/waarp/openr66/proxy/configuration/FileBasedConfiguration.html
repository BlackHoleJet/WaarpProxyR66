444 	private static boolean loadIdentity(Configuration config) {
445 		XmlValue value = hashConfig.get(XML_SERVER_HOSTID);
446 		if (value != null && (!value.isEmpty())) {
447 			config.HOST_ID = value.getString();
448 		} else {
449 			logger.error("Unable to find Host ID in Config file");
450 			return false;
451 		}
452 		value = hashConfig.get(XML_SERVER_SSLHOSTID);
453 		if (value != null && (!value.isEmpty())) {
454 			config.HOST_SSLID = value.getString();
455 		} else {
456 			logger
457 					.warn("Unable to find Host SSL ID in Config file so no SSL support will be used");
458 			config.useSSL = false;
459 			config.HOST_SSLID = null;
460 		}
461 		return setCryptoKey(config);
462 	}

464 	private static boolean loadServerParam(Configuration config) {
465 		XmlValue value = hashConfig.get(XML_USESSL);
466 		if (value != null && (!value.isEmpty())) {
467 			config.useSSL = value.getBoolean();
468 		}
469 		value = hashConfig.get(XML_USENOSSL);
470 		if (value != null && (!value.isEmpty())) {
471 			config.useNOSSL = value.getBoolean();
472 		}
473 		value = hashConfig.get(XML_USEHTTPCOMP);
474 		if (value != null && (!value.isEmpty())) {
475 			config.useHttpCompression = value.getBoolean();
476 		}
477 		value = hashConfig.get(XML_SERVER_ADMIN);
478 		if (value != null && (!value.isEmpty())) {
479 			config.ADMINNAME = value.getString();
480 		} else {
481 			logger.error("Unable to find Administrator name in Config file");
482 			return false;
483 		}
484 		if (config.cryptoKey == null) {
485 			if (!setCryptoKey(config)) {
486 				logger.error("Unable to find Crypto Key in Config file");
487 				return false;
488 			}
489 		}
490 		byte[] decodedByteKeys = null;
491 		value = hashConfig.get(XML_SERVER_PASSWD_FILE);
492 		if (value == null || (value.isEmpty())) {
493 			String passwd;
494 			value = hashConfig.get(XML_SERVER_PASSWD);
495 			if (value != null && (!value.isEmpty())) {
496 				passwd = value.getString();
497 			} else {
498 				logger.error("Unable to find Password in Config file");
499 				return false;
500 			}
501 			try {
502 				decodedByteKeys =
503 						config.cryptoKey.decryptHexInBytes(passwd);
504 			} catch (Exception e) {
505 				logger.error(
506 						"Unable to Decrypt Server Password in Config file from: " +
507 								passwd, e);
508 				return false;
509 			}
510 		} else {
511 			String skey = value.getString();
512 			// load key from file
513 			config.serverKeyFile = skey;
514 			File key = new File(skey);
515 			if (!key.canRead()) {
516 				logger.error("Unable to read Password in Config file from " + skey);
517 				return false;
518 			}
519 			try {
520 				decodedByteKeys = config.cryptoKey.decryptHexFile(key);
521 			} catch (Exception e2) {
522 				logger.error(
523 						"Unable to Decrypt Server Password in Config file from: " +
524 								skey, e2);
525 				return false;
526 			}
527 		}
528 		config.setSERVERKEY(decodedByteKeys);
529 		value = hashConfig.get(XML_HTTPADMINPATH);
530 		if (value == null || (value.isEmpty())) {
531 			logger.error("Unable to find Http Admin Base in Config file");
532 			return false;
533 		}
534 		String path = value.getString();
535 		if (path == null || path.isEmpty()) {
536 			logger.error("Unable to set correct Http Admin Base in Config file");
537 			return false;
538 		}
539 		File file = new File(path);
540 		if (!file.isDirectory()) {
541 			logger.error("Http Admin is not a directory in Config file");
542 			return false;
543 		}
544 		try {
545 			config.httpBasePath =
546 					FilesystemBasedDirImpl.normalizePath(file.getCanonicalPath()) +
547 							DirInterface.SEPARATOR;
548 		} catch (IOException e1) {
549 			logger.error("Unable to set Http Admin Path in Config file");
550 			return false;
551 		}
552 
553 		// Key for HTTPS
554 		value = hashConfig.get(XML_PATH_ADMIN_KEYPATH);
555 		if (value != null && (!value.isEmpty())) {
556 			String keypath = value.getString();
557 			if ((keypath == null) || (keypath.isEmpty())) {
558 				logger.error("Bad Key Path");
559 				return false;
560 			}
561 			value = hashConfig.get(XML_PATH_ADMIN_KEYSTOREPASS);
562 			if (value == null || (value.isEmpty())) {
563 				logger.error("Unable to find KeyStore Passwd");
564 				return false;
565 			}
566 			String keystorepass = value.getString();
567 			if ((keystorepass == null) || (keystorepass.isEmpty())) {
568 				logger.error("Bad KeyStore Passwd");
569 				return false;
570 			}
571 			value = hashConfig.get(XML_PATH_ADMIN_KEYPASS);
572 			if (value == null || (value.isEmpty())) {
573 				logger.error("Unable to find Key Passwd");
574 				return false;
575 			}
576 			String keypass = value.getString();
577 			if ((keypass == null) || (keypass.isEmpty())) {
578 				logger.error("Bad Key Passwd");
579 				return false;
580 			}
581 			try {
582 				Configuration.waarpSecureKeyStore =
583 						new WaarpSecureKeyStore(keypath, keystorepass,
584 								keypass);
585 			} catch (CryptoException e) {
586 				logger.error("Bad SecureKeyStore construction for AdminSsl");
587 				return false;
588 			}
589 			// No client authentication
590 			Configuration.waarpSecureKeyStore.initEmptyTrustStore();
591 			Configuration.waarpSslContextFactory =
592 					new WaarpSslContextFactory(
593 							Configuration.waarpSecureKeyStore, true);
594 		}
595 		value = hashConfig.get(XML_MONITOR_PASTLIMIT);
596 		if (value != null && (!value.isEmpty())) {
597 			config.pastLimit = (value.getLong() / 10) * 10;
598 		}
599 		value = hashConfig.get(XML_MONITOR_MINIMALDELAY);
600 		if (value != null && (!value.isEmpty())) {
601 			config.minimalDelay = (value.getLong() / 10) * 10;
602 		}
603 		value = hashConfig.get(XML_MONITOR_SNMP_CONFIG);
604 		if (value != null && (!value.isEmpty())) {
605 			config.snmpConfig = value.getString();
606 			File snmpfile = new File(config.snmpConfig);
607 			if (snmpfile.canRead()) {
608 				if (!SnmpConfiguration.setConfigurationFromXml(snmpfile)) {
609 					config.snmpConfig = null;
610 				}
611 			} else {
612 				config.snmpConfig = null;
613 			}
614 		}
615 		return true;
616 	}

618 	private static boolean loadDirectory(Configuration config) {

620 		if (value == null || (value.isEmpty())) {
