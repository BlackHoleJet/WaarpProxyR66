61  	@Override
62  	public void computeNbThreads() {
63  		int nb = Runtime.getRuntime().availableProcessors() + 1;
64  		SERVER_THREAD = nb;
65  		CLIENT_THREAD = SERVER_THREAD + 1;
66  		RUNNER_THREAD = 10;
67  	}

69  	@Override
70  	public void pipelineInit() {
71  		if (configured) {
72  			return;
73  		}
74  		InternalLoggerFactory.setDefaultFactory(InternalLoggerFactory
75  				.getDefaultFactory());
76  		objectSizeEstimator = new NetworkPacketSizeEstimator();
77  		httpPipelineInit();
78  		logger.warn("Server Thread: " + SERVER_THREAD + " Client Thread: " + CLIENT_THREAD
79  				+ " Runner Thread: " + RUNNER_THREAD);
80  		serverPipelineExecutor = new OrderedMemoryAwareThreadPoolExecutor(
81  				CLIENT_THREAD, maxGlobalMemory / 10, maxGlobalMemory, 1000,
82  				TimeUnit.MILLISECONDS, objectSizeEstimator,
83  				new WaarpThreadFactory("ServerExecutor"));
84  		configured = true;
85  	}

87  	@Override
88  	public void serverStartup() {
89  		isServer = true;
90  		shutdownConfiguration.timeout = TIMEOUTCON;
91  		R66ShutdownHook.addShutdownHook();
92  		if ((!useNOSSL) && (!useSSL)) {
93  			logger.error("OpenR66 has neither NOSSL nor SSL support included! Stop here!");
94  			System.exit(-1);
95  		}
96  		pipelineInit();
97  		r66Startup();
98  		startHttpSupport();
99  		try {
100 			startMonitoring();
101 		} catch (WaarpDatabaseSqlException e) {
102 		}
103 	}

105 	@Override
106 	public void r66Startup() {
107 		logger.debug("Start R66: " + SERVER_PORT + ":" + useNOSSL + " " + SERVER_SSLPORT + ":"
108 				+ useSSL + ":" + HOST_SSLID);
109 		// add into configuration
110 		this.constraintLimitHandler.setServer(true);
111 		// Global Server
112 		serverChannelGroup = new DefaultChannelGroup("OpenR66");
113 
114 		serverChannelFactory = new NioServerSocketChannelFactory(
115 				execServerBoss, execServerWorker, SERVER_THREAD);
116 		if (useNOSSL) {
117 			serverBootstrap = new ServerBootstrap(serverChannelFactory);
118 			networkServerPipelineFactory = new NetworkServerPipelineFactory(true);
119 			serverBootstrap.setPipelineFactory(networkServerPipelineFactory);
120 			serverBootstrap.setOption("child.tcpNoDelay", true);
121 			serverBootstrap.setOption("child.keepAlive", true);
122 			serverBootstrap.setOption("child.reuseAddress", true);
123 			serverBootstrap.setOption("child.connectTimeoutMillis", TIMEOUTCON);
124 			serverBootstrap.setOption("tcpNoDelay", true);
125 			serverBootstrap.setOption("reuseAddress", true);
126 			serverBootstrap.setOption("connectTimeoutMillis", TIMEOUTCON);
127 			// FIXME take into account multiple address
128 			for (ProxyEntry entry : ProxyEntry.proxyEntries.values()) {
129 				if (!entry.isLocalSsl()) {
130 					serverChannelGroup.add(serverBootstrap.bind(entry.getLocalSocketAddress()));
131 				}
132 			}
133 		} else {
134 			networkServerPipelineFactory = null;
135 			logger.warn("NOSSL mode is deactivated");
136 		}
137 
138 		if (useSSL && HOST_SSLID != null) {
139 			serverSslBootstrap = new ServerBootstrap(serverChannelFactory);
140 			networkSslServerPipelineFactory = new NetworkSslServerPipelineFactory(false,
141 					execServerWorker);
142 			serverSslBootstrap.setPipelineFactory(networkSslServerPipelineFactory);
143 			serverSslBootstrap.setOption("child.tcpNoDelay", true);
144 			serverSslBootstrap.setOption("child.keepAlive", true);
145 			serverSslBootstrap.setOption("child.reuseAddress", true);
146 			serverSslBootstrap.setOption("child.connectTimeoutMillis", TIMEOUTCON);
147 			serverSslBootstrap.setOption("tcpNoDelay", true);
148 			serverSslBootstrap.setOption("reuseAddress", true);
149 			serverSslBootstrap.setOption("connectTimeoutMillis", TIMEOUTCON);
150 			// FIXME take into account multiple address
151 			for (ProxyEntry entry : ProxyEntry.proxyEntries.values()) {
152 				if (entry.isLocalSsl()) {
153 					serverChannelGroup.add(serverSslBootstrap.bind(entry.getLocalSocketAddress()));
154 				}
155 			}
156 		} else {
157 			networkSslServerPipelineFactory = null;
158 			logger.warn("SSL mode is desactivated");
159 		}
160 
161 		// Factory for TrafficShapingHandler
162 		globalTrafficShapingHandler = new GlobalTrafficHandler(
163 				objectSizeEstimator, timerTrafficCounter,
164 				serverGlobalWriteLimit, serverGlobalReadLimit, delayLimit);
165 		this.constraintLimitHandler.setHandler(globalTrafficShapingHandler);
166 		ProxyBridge.initialize();
167 		localTransaction = new LocalTransaction();
168 		thriftService = null;
169 	}

171 	@Override
172 	public void startHttpSupport() {
173 		// Now start the HTTP support
174 		httpChannelGroup = new DefaultChannelGroup("HttpOpenR66");
175 		// Configure the server.
176 		httpChannelFactory = new NioServerSocketChannelFactory(
177 				execServerBoss,
178 				execServerWorker,
179 				SERVER_THREAD);
180 		httpBootstrap = new ServerBootstrap(
181 				httpChannelFactory);
182 		// Set up the event pipeline factory.
183 		httpBootstrap.setPipelineFactory(new HttpPipelineFactory(useHttpCompression));
184 		httpBootstrap.setOption("child.tcpNoDelay", true);
185 		httpBootstrap.setOption("child.keepAlive", true);
186 		httpBootstrap.setOption("child.reuseAddress", true);
187 		httpBootstrap.setOption("child.connectTimeoutMillis", TIMEOUTCON);
188 		httpBootstrap.setOption("tcpNoDelay", true);
189 		httpBootstrap.setOption("reuseAddress", true);
190 		httpBootstrap.setOption("connectTimeoutMillis", TIMEOUTCON);
191 		// Bind and start to accept incoming connections.
192 		httpChannelGroup.add(httpBootstrap.bind(new InetSocketAddress(SERVER_HTTPPORT)));
193 
194 		// Now start the HTTPS support
195 		// Configure the server.
196 		httpsChannelFactory = new NioServerSocketChannelFactory(
197 				execServerBoss,
198 				execServerWorker,
199 				SERVER_THREAD);
200 		httpsBootstrap = new ServerBootstrap(
201 				httpsChannelFactory);
202 		// Set up the event pipeline factory.
203 		httpsBootstrap.setPipelineFactory(new HttpSslPipelineFactory(useHttpCompression,
204 				false, execServerWorker));
205 		httpsBootstrap.setOption("child.tcpNoDelay", true);
206 		httpsBootstrap.setOption("child.keepAlive", true);
207 		httpsBootstrap.setOption("child.reuseAddress", true);
208 		httpsBootstrap.setOption("child.connectTimeoutMillis", TIMEOUTCON);
209 		httpsBootstrap.setOption("tcpNoDelay", true);
210 		httpsBootstrap.setOption("reuseAddress", true);
211 		httpsBootstrap.setOption("connectTimeoutMillis", TIMEOUTCON);
212 		// Bind and start to accept incoming connections.
213 		httpChannelGroup.add(httpsBootstrap.bind(new InetSocketAddress(SERVER_HTTPSPORT)));
214 	}
